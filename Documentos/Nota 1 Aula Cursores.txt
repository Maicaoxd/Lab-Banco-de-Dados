

 CURSORES ( DEVOLVE UM PONTEIRO DO PRIMEIRO REGISTRO DO CONJUNTO RESULTADO)



 1-  EXPLICITO

 		DECLARE CURSOR C1 
 		OPEN C1
 		LOOP
 			FETCH
 		END LOOP

 		CLOSE C1  (NAO ESQUECER DISSO)



 2 - IMPLICITO

 		"OPEN FETCH E CLOSE", SAO EXECUTADOS IMPLICITAMENTE DENTRO DE UM COMANDO FOR/LOOP. (EXCLUSIVO ORACLE)

QUANDO SE DESEJA LISTAR TODOS OS REGISTROS DO CONJUNTO RESULTADO.



- EXEMPLO PARA MYSQL DE CURSOR

DELIMITER $$  -- SIGNIFICA FINAL DO PROCEDIMENTO
CREATE PROCEDIMENTO cursor_exemplo()
BEGIN
	DECLARE  nome VARCHAR(60);
	DECLARE done INT DEFAULT 0;
	DECLARE C1 CURSOR FOR 
			select nome_disciplina from disciplina ;
	DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = 1;
	OPEN C1;
	WHILE done != 1 DO
		FETCH c1 into nome;
		SELECT nome;
	END WHILE;
	CLOSE c1;
	END$$
